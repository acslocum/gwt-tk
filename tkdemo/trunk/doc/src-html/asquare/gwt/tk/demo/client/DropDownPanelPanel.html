016    package asquare.gwt.tk.demo.client;
017    
018    import asquare.gwt.tk.client.ui.BasicPanel;
019    import asquare.gwt.tk.client.ui.DropDownPanel;
020    import asquare.gwt.tk.client.ui.SimpleHyperLink;
021    
022    import com.google.gwt.user.client.DOM;
023    import com.google.gwt.user.client.ui.*;
024    
025    public class DropDownPanelPanel extends Composite
026    {
027            public DropDownPanelPanel()
028            {
029                    HorizontalPanel outer = new HorizontalPanel();
030                    outer.setSize("100%", "100%");
031    
032                    setWidget(outer);
033    
034                    BasicPanel left = new BasicPanel();
035                    DOM.setAttribute(left.getElement(), "id", "dropDown-left");             
036                    
037                    DropDownPanel dropDown1 = new DropDownPanel();
038                    DOM.setAttribute(dropDown1.getElement(), "id", "dropDown-1");           
039                    dropDown1.setHeaderText("Drop Down 1", false);
040                    Image image = new Image("icecube.jpg");
041                    image.setSize("100px", "100px");
042                    dropDown1.add(image);
043                    dropDown1.setOpen(true);
044                    left.add(dropDown1);
045                    
046                    DropDownPanel dropDown2 = new DropDownPanel();
047                    DOM.setAttribute(dropDown2.getElement(), "id", "dropDown-2");           
048                    dropDown2.setHeaderText("Drop Down 2", true);
049                    image = new Image("icecube.jpg");
050                    image.setSize("100px", "250px");
051                    dropDown2.add(image);
052                    left.add(dropDown2);
053                    
054                    outer.add(left);
055                    outer.setCellWidth(left, "1px");
056                    
057                    String content = 
058                            "<H2>Drop Down Panel</H2>" + 
059                            "<p>This widget consists of a hideable content DIV and an optional header DIV.</p>" + 
060                            "<p>The content can be <span id='open'></span> & <span id='close'></span> programatically.</p>" + 
061                            "<p>The headers prevent selection of the text within them. In IE this is done in a special implementation using an <code>onselectstart</code> listener which returns <code>false</code>. In other browsers, <code>DOM.eventPreventDefault(Event)</code> for <code>Event.ONMOUSEDOWN</code> suffices. </p>" +
062                            "<p>You can add a listener to the DropDownPanel and make layout changes when the it opens/closes (e.g. maximize its parent element when it opens). </p>" + 
063                            "<p>The tip-down arrows on the left are background images applied through CSS rules. <em>Note: we cannot put a border around the header itself because in IE the background image would be obscured by the border.</em></p>" +
064                            "<h4>Known problems</h4>" +
065                            "<ul><li>In IE, the cursor reverts to the system arrow after opening the headers on the left. This is a result of changing the tipdown arrow background image.</li></ul>" + 
066                            "<p>Tested in IE 6 (Win), Firefox 1.5 (Win), Opera 8 (win) and Safari 2.0.4 (Mac) in Strict and Quirks modes.</p>";
067                    DropDownPanel dropDown3 = new DropDownPanel();
068                    DOM.setAttribute(dropDown3.getElement(), "id", "dropDown-3");           
069                    dropDown3.setHeaderText("Drop Down 3", true);
070                    HTMLPanel center = new HTMLPanel(content);
071                    SimpleHyperLink close = new SimpleHyperLink("hidden");
072                    SimpleHyperLink open = new SimpleHyperLink("shown");
073                    center.add(close, "close");
074                    center.add(open, "open");
075                    dropDown3.add(center);
076                    dropDown3.setOpen(true);
077                    outer.add(dropDown3);
078                    
079                    final DropDownPanel dropDown4 = new DropDownPanel();
080                    DOM.setAttribute(dropDown4.getElement(), "id", "dropDown-4");
081                    image = new Image("icecube.jpg");
082                    image.setSize("200px", "250px");
083                    dropDown4.add(image);
084                    outer.add(dropDown4);
085                    outer.setCellWidth(dropDown4, "1px");
086                    
087                    open.addClickListener(new ClickListener()
088                    {
089                            public void onClick(Widget sender)
090                            {
091                                    dropDown4.setOpen(true);
092                            }
093                    });
094                    close.addClickListener(new ClickListener()
095                    {
096                            public void onClick(Widget sender)
097                            {
098                                    dropDown4.setOpen(false);
099                            }
100                    });
101            }
102    }
