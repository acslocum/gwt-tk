033            public DebugPanel()
034            {
035                    HorizontalPanel outer = new HorizontalPanel();
036                    
037                    setWidget(outer);

040                    outer.add(docPanel);

051            }

Press <code>w</code> twice to enable/disable the debug console independently of other debug funtions.</p>" +

104                    }));

106                    {

137                    
138                    return controls;
139            }

141            private static class EventToMask
142            {
143                    public final String m_event;
144                    public final int m_mask;
145                    
146                    public EventToMask(String event, int mask)
147                    {
148                            m_event = event;
149                            m_mask = mask;
150                    }
151    
152                    public boolean equals(Object obj)
153                    {
154                            return m_event.equals(obj);
155                    }
156            }
157            
158            private Widget createCheckBoxes()
159            {
160                    // can't use HashMap because it does not preserve order
161                    final Vector eventToMask = new Vector();
162                    eventToMask.add(new EventToMask("onfocus", Event.ONFOCUS));
163                    eventToMask.add(new EventToMask("onblur", Event.ONBLUR));
164                    eventToMask.add(new EventToMask("onchange", Event.ONCHANGE));
165                    eventToMask.add(new EventToMask("onkeydown", Event.ONKEYDOWN));
166                    eventToMask.add(new EventToMask("onkeypress", Event.ONKEYPRESS));
167                    eventToMask.add(new EventToMask("onkeyup", Event.ONKEYUP));
168                    eventToMask.add(new EventToMask("onscroll", Event.ONSCROLL));
169                    eventToMask.add(new EventToMask("onmousemove", Event.ONMOUSEMOVE));
170                    eventToMask.add(new EventToMask("onmouseover", Event.ONMOUSEOVER));
171                    eventToMask.add(new EventToMask("onmouseout", Event.ONMOUSEOUT));
172                    eventToMask.add(new EventToMask("onmousedown", Event.ONMOUSEDOWN));
173                    eventToMask.add(new EventToMask("onmouseup", Event.ONMOUSEUP));
174                    eventToMask.add(new EventToMask("onclick", Event.ONCLICK));
175                    eventToMask.add(new EventToMask("ondblclick", Event.ONDBLCLICK));
176                    eventToMask.add(new EventToMask("onerror", Event.ONERROR));
177                    
178                    ClickListener checkBoxController = new ClickListener()
179                    {
180                            public void onClick(Widget sender)
181                            {
182                                    CheckBox cb = (CheckBox) sender;
183                                    EventToMask mapping = (EventToMask) eventToMask.get(eventToMask.indexOf(cb.getText()));
184                                    int traceEventMask = m_eventListener.getEventMask();
185                                    if (cb.isChecked())
186                                    {
187                                            traceEventMask |= mapping.m_mask;
188                                    }
189                                    else
190                                    {
191                                            traceEventMask &= ~mapping.m_mask;
192                                    }
193                                    m_eventListener.setEventMask(traceEventMask);
194                            }
195                    };
196                    
197                    FlowPanel panel = new FlowPanel();
198                    for (int i = 0; i < eventToMask.size(); i++)
199                    {
200                            EventToMask mapping = (EventToMask) eventToMask.get(i);
201                            addCheckBox(panel, mapping.m_event, mapping.m_mask, checkBoxController);
202                    }
203                    return panel;
204            }
205            
206            private void addCheckBox(ComplexPanel parent, String label, int mask, ClickListener listener)
207            {
208                    CheckBox cb = new CheckBox(label);
209                    cb.setChecked((m_eventListener.getEventMask() & mask) != 0);
210                    cb.addClickListener(listener);
211                    parent.add(cb);
212                    DOM.appendChild(parent.getElement(), DOM.createElement("br"));
213            }

215            private Widget createWidgetPanel()
216            {
217                    BasicPanel widgets = new BasicPanel();
218                    DOM.setAttribute(widgets.getElement(), "id", "debug-widgets");
219                    TextArea textArea = new TextArea();
220                    textArea.setText("Enable event tracing then type in here");
221                    widgets.add(textArea);
222                    
223                    ListBox listBox = new ListBox();
224                    listBox.addItem("List Box");
225                    listBox.addItem("foo");
226                    listBox.addItem("bar");
227                    listBox.addItem("baz");
228                    widgets.add(listBox);
229                    
230                    TreeItem treeRoot = new TreeItem("Tree");
231                    TreeItem treeItem = new TreeItem("foo");
232                    treeRoot.addItem(treeItem);
233                    treeRoot.addItem("bar");
234                    treeRoot.addItem("baz");
235                    Tree tree = new Tree();
236                    tree.addItem(treeRoot);
237                    tree.setSelectedItem(treeItem);
238                    tree.ensureSelectedItemVisible();
239                    widgets.add(tree);
240                    
241                    Image image = new Image("icecube.jpg");
242                    ScrollPanel scrollPanel = new ScrollPanel(image);
243                    scrollPanel.setSize("200px", "200px");
244                    widgets.add(scrollPanel);
245                    
246                    return widgets;
247            }
